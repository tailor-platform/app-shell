import { ActionPanel } from "@tailor-platform/app-shell";

Use `ActionPanel` to display a vertical list of clickable actions with icons. It is designed for ERP detail pages and side panels where available actions (e.g. "Create invoice", "Export CSV") are listed alongside a record.

## Basic

```tsx preview
<div style={{ width: 350, padding: "16px 0" }}>
  <ActionPanel
    title="Actions"
    actions={[
      { key: "edit", label: "Edit record", icon: <span>✏️</span>, onClick: () => {} },
      { key: "copy", label: "Duplicate", icon: <span>📋</span>, onClick: () => {} },
      { key: "delete", label: "Delete", icon: <span>🗑️</span>, onClick: () => {} },
    ]}
  />
</div>
```

## Loading State

```tsx preview
<div style={{ width: 350, padding: "16px 0" }}>
  <ActionPanel
    title="Processing"
    actions={[
      { key: "save", label: "Saving…", icon: <span>💾</span>, loading: true },
      { key: "export", label: "Export CSV", icon: <span>📤</span>, onClick: () => {} },
    ]}
  />
</div>
```

## Disabled

<div style={{ width: 350, padding: "16px 0" }}>
