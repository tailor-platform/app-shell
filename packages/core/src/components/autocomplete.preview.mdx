import { Autocomplete } from "@tailor-platform/app-shell";

Use `Autocomplete` when users should type free-form text but benefit from suggestions. Unlike `Select` or `Combobox`, the value is the raw input string — not a discrete selection. Ideal for search fields, address inputs, or any input where suggestions assist but don't constrain the user.

## Basic

```tsx preview
<div style={{ width: 400 }}>
  <Autocomplete
    items={["Tokyo", "Osaka", "Kyoto", "Nagoya", "Fukuoka", "Sapporo"]}
    placeholder="Search cities..."
  />
</div>
```

## With Objects

```tsx preview
<div style={{ width: 400 }}>
  <Autocomplete
    items={[
      { code: "US", name: "United States" },
      { code: "JP", name: "Japan" },
      { code: "GB", name: "United Kingdom" },
      { code: "DE", name: "Germany" },
      { code: "FR", name: "France" },
    ]}
    mapItem={(item) => ({ label: item.name, key: item.code })}
    placeholder="Search countries..."
  />
</div>
```

## Grouped Items

```tsx preview
<div style={{ width: 400 }}>
  <Autocomplete
    items={[
      { label: "Asia", items: ["Tokyo", "Seoul", "Bangkok"] },
      { label: "Europe", items: ["London", "Paris", "Berlin"] },
    ]}
    placeholder="Search cities..."
  />
</div>
```

## Disabled

```tsx preview
<div style={{ width: 400 }}>
  <Autocomplete items={["Tokyo", "Osaka", "Kyoto"]} placeholder="Disabled autocomplete" disabled />
</div>
```
