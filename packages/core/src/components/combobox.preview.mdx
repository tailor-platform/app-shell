import { Combobox } from "@tailor-platform/app-shell";

Use `Combobox` when users need to search and select from a list of options. It combines a text input with a filterable dropdown, making it ideal for large option sets. Supports multi-select, async loading, and user-created items.

## Basic

```tsx preview
<div style={{ width: 400 }}>
  <Combobox
    items={["Apple", "Banana", "Cherry", "Date", "Elderberry", "Fig", "Grape"]}
    placeholder="Search fruits..."
  />
</div>
```

## With Objects

```tsx preview
<div style={{ width: 400 }}>
  <Combobox
    items={[
      { id: 1, name: "React" },
      { id: 2, name: "Vue" },
      { id: 3, name: "Angular" },
      { id: 4, name: "Svelte" },
    ]}
    mapItem={(item) => ({ label: item.name, key: String(item.id) })}
    placeholder="Search frameworks..."
  />
</div>
```

## Multiple

```tsx preview
<div style={{ width: 400 }}>
  <Combobox
    items={["React", "Vue", "Angular", "Svelte", "Solid", "Preact"]}
    placeholder="Select frameworks..."
    multiple
  />
</div>
```

## Grouped Items

```tsx preview
<div style={{ width: 400 }}>
  <Combobox
    items={[
      { label: "Frontend", items: ["React", "Vue", "Angular"] },
      { label: "Backend", items: ["Express", "Fastify", "Hono"] },
    ]}
    placeholder="Search frameworks..."
  />
</div>
```

## Disabled

```tsx preview
<div style={{ width: 400 }}>
  <Combobox items={["Apple", "Banana", "Cherry"]} placeholder="Disabled combobox" disabled />
</div>
```
