import { DescriptionCard } from "@tailor-platform/app-shell";

Use `DescriptionCard` to display structured key-value information for a record — such as customer details, order summaries, or document metadata. It renders field-value pairs in a responsive grid with built-in support for badges, monetary values, dates, and links.

## Basic

The simplest usage with plain text fields. Pass `data` as a key-value object and `fields` to define which keys to display and their labels.

```tsx preview
<div style={{ width: "100%" }}>
  <DescriptionCard
    title="Customer"
    data={{
      name: "Acme Corp",
      email: "contact@acme.com",
      phone: "+1-555-0123",
    }}
    fields={[
      { key: "name", label: "Name" },
      { key: "email", label: "Email" },
      { key: "phone", label: "Phone" },
    ]}
  />
</div>
```

## With Badge and Money

Set `type` on a field to render rich content. `badge` maps values to colored badges via `badgeVariantMap`, `money` formats numbers as currency using a sibling `currencyKey`, and `date` formats ISO strings with the specified `dateFormat`.

```tsx preview
<div style={{ width: "100%" }}>
  <DescriptionCard
    title="Order #1234"
    data={{
      status: "CONFIRMED",
      total: 1250.0,
      currency: "USD",
      createdAt: "2025-06-15T10:30:00Z",
    }}
    fields={[
      {
        key: "status",
        label: "Status",
        type: "badge",
        meta: { badgeVariantMap: { CONFIRMED: "success", PENDING: "warning", CANCELLED: "error" } },
      },
      { key: "total", label: "Total", type: "money", meta: { currencyKey: "currency" } },
      { key: "createdAt", label: "Created", type: "date", meta: { dateFormat: "medium" } },
    ]}
  />
</div>
```

## With Divider

Insert `{ type: "divider" }` in the `fields` array to visually separate groups of related fields.

```tsx preview
<div style={{ width: "100%" }}>
  <DescriptionCard
    title="Shipping Info"
    data={{
      recipient: "John Doe",
      address: "123 Main St, Springfield",
      carrier: "FedEx",
      tracking: "FX123456789",
    }}
    fields={[
      { key: "recipient", label: "Recipient" },
      { key: "address", label: "Address" },
      { type: "divider" },
      { key: "carrier", label: "Carrier" },
      { key: "tracking", label: "Tracking #", meta: { copyable: true } },
    ]}
  />
</div>
```

## 4 Columns

Use the `columns` prop to control the grid layout. The default is 3 columns; here 4 columns are used to display more fields per row.

```tsx preview
<div style={{ width: "100%" }}>
  <DescriptionCard
    title="Product Details"
    columns={4}
    data={{
      sku: "PROD-001",
      name: "Widget Pro",
      category: "Electronics",
      price: 49.99,
      currency: "USD",
      weight: "0.5 kg",
      stock: "In Stock",
      manufacturer: "WidgetCo",
    }}
    fields={[
      { key: "sku", label: "SKU", meta: { copyable: true } },
      { key: "name", label: "Product Name" },
      { key: "category", label: "Category" },
      { key: "price", label: "Price", type: "money", meta: { currencyKey: "currency" } },
      { key: "weight", label: "Weight" },
      {
        key: "stock",
        label: "Availability",
        type: "badge",
        meta: { badgeVariantMap: { "In Stock": "success" } },
      },
      { key: "manufacturer", label: "Manufacturer" },
    ]}
  />
</div>
```
