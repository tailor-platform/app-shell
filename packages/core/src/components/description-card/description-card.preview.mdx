## With Badge and Money

<div style={{ padding: "16px 0" }}>
  <DescriptionCard
    title="Order #1234"
    data={{
      status: "CONFIRMED",
      total: 1250.0,
      currency: "USD",
      createdAt: "2025-06-15T10:30:00Z",
    }}
    fields={[
      {
        key: "status",
        label: "Status",
        type: "badge",
        meta: { badgeVariantMap: { CONFIRMED: "success", PENDING: "warning", CANCELLED: "error" } },

      { key: "total", label: "Total", type: "money", meta: { currencyKey: "currency" } },
      { key: "createdAt", label: "Created", type: "date", meta: { dateFormat: "medium" } },

  />
</div>

<div style={{ padding: "16px 0" }}>
  <DescriptionCard
    title="Product Details"
    columns={4}
    data={{
      sku: "PROD-001",
      name: "Widget Pro",
      category: "Electronics",
      price: 49.99,
      currency: "USD",
      weight: "0.5 kg",
      stock: "In Stock",
      manufacturer: "WidgetCo",
    }}
    fields={[
      { key: "sku", label: "SKU", meta: { copyable: true } },
      { key: "name", label: "Product Name" },
      { key: "category", label: "Category" },
      { key: "price", label: "Price", type: "money", meta: { currencyKey: "currency" } },
      { key: "weight", label: "Weight" },
      {
        key: "stock",
        label: "Availability",
        type: "badge",
        meta: { badgeVariantMap: { "In Stock": "success" } },
      },
      { key: "manufacturer", label: "Manufacturer" },
    ]}
  />
</div>
