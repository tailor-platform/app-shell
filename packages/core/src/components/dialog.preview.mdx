import { Dialog, Button } from "@tailor-platform/app-shell";

## Basic

<div style={{ padding: "16px 0" }}>
  <Dialog.Root>
    <Dialog.Trigger render={<Button variant="outline" />}>Open Dialog</Dialog.Trigger>
    <Dialog.Content>
      <Dialog.Header>
        <Dialog.Title>Are you sure?</Dialog.Title>
        <Dialog.Description>This action cannot be undone. This will permanently delete your account and remove your data from our servers.</Dialog.Description>
      </Dialog.Header>
      <Dialog.Footer>
        <Dialog.Close render={<Button variant="outline" />}>Cancel</Dialog.Close>
        <Button>Confirm</Button>
      </Dialog.Footer>
    </Dialog.Content>
  </Dialog.Root>
</div>

```tsx
<Dialog.Root>
  <Dialog.Trigger render={<Button variant="outline" />}>Open Dialog</Dialog.Trigger>
```
