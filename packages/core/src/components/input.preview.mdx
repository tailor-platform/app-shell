import { Input } from "@tailor-platform/app-shell";

Use `Input` for single-line text entry — names, emails, passwords, search queries, and other short-form values. It wraps a native `<input>` with consistent theming and supports all standard HTML input attributes.

## Basic

```tsx preview
<div style={{ display: "flex", flexDirection: "column", gap: 8, width: 400 }}>
  <Input type="text" placeholder="Enter your name" />
  <Input type="email" placeholder="Email address" />
  <Input type="password" placeholder="Password" />
</div>
```

## Disabled

```tsx preview
<div style={{ width: 400 }}>
  <Input placeholder="Disabled input" disabled />
</div>
```

## File Input

```tsx preview
<div style={{ width: 400 }}>
  <Input type="file" />
</div>
```
