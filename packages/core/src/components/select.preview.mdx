import { Select } from "@tailor-platform/app-shell";

Use `Select` when users need to pick one or more options from a predefined list without typing. It provides a dropdown with single or multi-select and supports grouped items and async data fetching.

## Basic

```tsx preview
<div style={{ width: 400 }}>
  <Select
    items={["Apple", "Banana", "Cherry", "Date", "Elderberry"]}
    placeholder="Select a fruit"
  />
</div>
```

## With Objects

```tsx preview
<div style={{ width: 400 }}>
  <Select
    items={[
      { id: 1, name: "Active" },
      { id: 2, name: "Inactive" },
      { id: 3, name: "Pending" },
    ]}
    mapItem={(item) => ({ label: item.name, key: String(item.id) })}
    placeholder="Select a status"
  />
</div>
```

## Grouped Items

```tsx preview
<div style={{ width: 400 }}>
  <Select
    items={[
      { label: "Fruits", items: ["Apple", "Banana", "Cherry"] },
      { label: "Vegetables", items: ["Carrot", "Broccoli", "Spinach"] },
    ]}
    placeholder="Select a food"
  />
</div>
```

## Multiple

```tsx preview
<div style={{ width: 400 }}>
  <Select
    items={["Red", "Green", "Blue", "Yellow", "Purple"]}
    placeholder="Select colors"
    multiple
  />
</div>
```

## Disabled

```tsx preview
<div style={{ width: 400 }}>
  <Select items={["Apple", "Banana", "Cherry"]} placeholder="Disabled select" disabled />
</div>
```
